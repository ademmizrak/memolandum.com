    loadLevelChunk() {
        const start = this.chunkIndex * this.wordsPerLevel;
        const chunkWords = this.vocabulary.slice(start, start + this.wordsPerLevel);
        if (chunkWords.length === 0) {
            return false;
        }
        this.activeChunk = chunkWords.map((w)=>({
                ...w,
                hitCount: 0
            }));
        return true;
    }
    setupBrickGrid() {
        // 12 words without duplication = 12 bricks
        const items = [];
        this.activeChunk.forEach((word)=>{
            items.push({
                ...word
            });
        });
        // Shuffle items to distribute randomly
        items.sort(()=>0.5 - Math.random());
        const cols = 4;
        const rows = Math.ceil(items.length / cols);
        const brickWidth = 120;
        const brickHeight = 36;
        const colSpacing = 20;
        const rowSpacing = 16;
        const totalGridWidth = cols * brickWidth + (cols - 1) * colSpacing;
        const startX = (this.virtualWidth - totalGridWidth) / 2;
        const startY = 150;
        this.activeBricks = [];
        let itemIdx = 0;
        for(let r = 0; r < rows; r++){
            for(let c = 0; c < cols; c++){
                if (itemIdx < items.length) {
                    const bx = startX + c * (brickWidth + colSpacing);
                    const by = startY + r * (brickHeight + rowSpacing);
                    const brick = this.brickPool.acquire(bx, by, brickWidth, brickHeight, items[itemIdx]);
                    this.activeBricks.push(brick);
                    itemIdx++;
                }
            }
        }
    }
    spawnBallOnPaddle() {

        const speed = 400 * this.speedMultiplier;
        const vx = speed * Math.sin(angle);
        const vy = -speed * Math.cos(angle);
        const ball = this.ballPool.acquire(bx, by, vx, vy);
        this.activeBalls.push(ball);
    }

    gameLoop(currentTime) {
        if (window.game !== this) return;
        this.loopId = requestAnimationFrame((time)=>this.gameLoop(time));
        if (!this.lastTime) this.lastTime = Math.max(0, currentTime - 16.6);
        let dt = (currentTime - this.lastTime) / 1000;
        this.lastTime = currentTime;
        if (dt > 0.1) dt = 0.1;
        if (this.state !== 'playing' && this.state !== 'exam') {
            this.stars.forEach((star)=>star.update(dt, this.virtualWidth, this.virtualHeight));
            this.draw();
            return;
        }
        if (this.isPaused) {
            this.draw();
            return;
        }
        if (performance.now() < this.hitPauseUntil) {
            this.draw();
            return;
        }
        this.update(dt);
        this.draw();
    }
    update(dt) {
        if (this.state === 'exam') {
            if (window.examEngine) {
                window.examEngine.update(dt);
            }
            return;
        }
        // Update background stars
        this.stars.forEach((star)=>star.update(dt, this.virtualWidth, this.virtualHeight));
        // Level transition countdown
        if (this.isLevelTransitioning) {
            this.levelCompleteTimer -= dt;
            // Update particles
            for(let i = this.activeParticles.length - 1; i >= 0; i--){
                const p = this.activeParticles[i];
                p.update(dt);
                if (!p.active) {
                    this.particlePool.release(p);
                    this.activeParticles.splice(i, 1);
                }
            }
            if (this.levelCompleteTimer <= 1.2 && !this.riserPlayed) {
                this.soundManager.playStageRiser();
                this.riserPlayed = true;
            }
            if (this.levelCompleteTimer <= 0) {
                this.isLevelTransitioning = false;
                // Progress to next chunk
                this.chunkIndex++;
                const hasNext = this.loadLevelChunk();
                if (hasNext) {
                    this.setupBrickGrid();
                    this.spawnBallOnPaddle();
                    this.updateHUD();
                    localStorage.setItem('memolandum_saved_level', this.chunkIndex + 1);
                    localStorage.setItem('memolandum_saved_score', this.score);
                // removed
                } else {
                    this.triggerFinalVictory();
                }
            }
            return;
        }
        // Keyboard paddle movement
        const paddleSpeed = 500 * this.speedMultiplier;
        if (this.input.left) {
            this.paddle.x -= paddleSpeed * dt;
            this.constrainPaddle();
        }
        if (this.input.right) {
            this.paddle.x += paddleSpeed * dt;
            this.constrainPaddle();
        }
        // Update active balls
        for(let i = this.activeBalls.length - 1; i >= 0; i--){
            const ball = this.activeBalls[i];
            ball.update(dt);
            // Wall bounces
            if (ball.x - ball.radius < 0) {
                ball.x = ball.radius;
                ball.vx = -ball.vx;
                this.soundManager.playGemTick();
            }
            if (ball.x + ball.radius > this.virtualWidth) {
                ball.x = this.virtualWidth - ball.radius;
                ball.vx = -ball.vx;
                this.soundManager.playGemTick();
            }
            if (ball.y - ball.radius < 80) {
                ball.y = ball.radius + 80;
                ball.vy = -ball.vy;
                this.soundManager.playGemTick();
            }
            // Paddle vector collision bounce
            if (ball.vy > 0 && ball.y + ball.radius >= this.paddle.y && ball.y - ball.radius <= this.paddle.y + this.paddle.height && ball.x + ball.radius >= this.paddle.x - this.paddle.width / 2 && ball.x - ball.radius <= this.paddle.x + this.paddle.width / 2) {
                ball.y = this.paddle.y - ball.radius;
                const relativeHit = (ball.x - this.paddle.x) / (this.paddle.width / 2);
                const maxAngle = Math.PI / 3;
                const angle = relativeHit * maxAngle;
                const speed = 400 * this.speedMultiplier;
                ball.vx = speed * Math.sin(angle);
                ball.vy = -speed * Math.cos(angle);
                this.soundManager.playGemTick();
                this.screenShake = 2;
            }
            // Ball brick collision checks
            for(let j = this.activeBricks.length - 1; j >= 0; j--){
                const brick = this.activeBricks[j];
                if (brick.active && checkBallBrickCollision(ball, brick)) {
                    brick.hp--;
                    if (brick.hp <= 0) {
                        brick.active = false;
                        this.soundManager.playExplosion();
                        this.spawnExplosion(brick.x + brick.width / 2, brick.y + brick.height / 2, '#00f0ff');
                        this.screenShake = 4;
                        // Spawn floating feedback text
                        const ft = this.floatingTextPool.acquire(brick.x + brick.width / 2, brick.y + brick.height / 2, brick.word.turkish);
                        this.activeFloatingTexts.push(ft);
                        const wordEnglish = brick.word.english;
                        if (this.playAudioCallback) this.playAudioCallback(wordEnglish);
                        const remainingBricks = this.activeBricks.filter((b)=>b.active && b.word.english === wordEnglish).length;
                        if (remainingBricks === 0) {
                            brick.word.hitCount = 2;
                            if (!this.wordsLearnedThisRun.some((w)=>w.english === wordEnglish)) {
                                this.wordsLearnedThisRun.push(brick.word);
                            }
                            if (window.examEngine) {
                                window.examEngine.registerAttempt(wordEnglish, true);
                            }
                            this.score += 10;
                            this.updateHUD();
                        }
                        // Level complete check
                        const allCleared = this.activeBricks.every((b)=>!b.active);
                        if (allCleared) {
                            // Burst 10 gems from center of last explosion
                            const brickCX = brick.x + brick.width / 2;
                            const brickCY = brick.y + brick.height / 2;
                            for(let gIdx = 0; gIdx < 10; gIdx++){
                                const gem = this.gemPool.acquire(brickCX, brickCY);
                                this.activeGems.push(gem);
                            }
                            this.triggerLevelComplete();
                            return;
                        }
                    } else {
                        this.soundManager.playGemTick();
                    }
                    break;
                }
            }
            // Out of bounds drop
            if (ball.y - ball.radius > 930) {
                this.ballPool.release(ball);
                this.activeBalls.splice(i, 1);
                if (this.activeBalls.length === 0) {
                    this.decreaseShields();
                    // Register database attempt failures for all active remaining words
                    const activeWords = new Set(this.activeBricks.filter((b)=>b.active).map((b)=>b.word.english));
                    activeWords.forEach((wordEnglish)=>{
                        if (window.examEngine) {
                            window.examEngine.registerAttempt(wordEnglish, false);
                        }
                    });
                    if (this.shields > 0) {
                        this.spawnBallOnPaddle();
                    }
                }
            }
        }
        // Update active gems
        for(let i = this.activeGems.length - 1; i >= 0; i--){
            const gem = this.activeGems[i];
            gem.update(dt);
            if (gem.active && gem.y >= this.paddle.y && gem.y <= this.paddle.y + this.paddle.height && gem.x >= this.paddle.x - this.paddle.width / 2 && gem.x <= this.paddle.x + this.paddle.width / 2) {
                gem.active = false;
                this.collectGem();
            }
            if (!gem.active) {
                this.gemPool.release(gem);
                this.activeGems.splice(i, 1);
            }
        }
        // Update floating translation texts
        for(let i = this.activeFloatingTexts.length - 1; i >= 0; i--){
            const ft = this.activeFloatingTexts[i];
            ft.update(dt);
            if (!ft.active) {
                this.floatingTextPool.release(ft);
                this.activeFloatingTexts.splice(i, 1);
            }
        }
        // Update explosion particles
        for(let i = this.activeParticles.length - 1; i >= 0; i--){
            const p = this.activeParticles[i];
            p.update(dt);
            if (!p.active) {
                this.particlePool.release(p);
                this.activeParticles.splice(i, 1);
            }
        }
        if (this.screenShake > 0) {
            this.screenShake -= dt * 20;
            if (this.screenShake < 0) this.screenShake = 0;
        }
    }
    draw() {
        if (this.state === 'exam') {
            this.ctx.fillStyle = '#000000';
            this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
            this.ctx.save();
            if (this.screenShake > 0) {
                const dx = (Math.random() - 0.5) * this.screenShake;
                const dy = (Math.random() - 0.5) * this.screenShake;
                this.ctx.translate(dx, dy);
            }
            this.ctx.scale(this.scaleX, this.scaleY);
            if (window.examEngine) {
                window.examEngine.draw(this.ctx);
            }
            this.ctx.restore();
            return;
        }
        this.ctx.fillStyle = '#000000';
        this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
        this.ctx.save();
        if (this.screenShake > 0) {
            const dx = (Math.random() - 0.5) * this.screenShake;
            const dy = (Math.random() - 0.5) * this.screenShake;
            this.ctx.translate(dx, dy);
        }
        this.ctx.scale(this.scaleX, this.scaleY);
        this.stars.forEach((star)=>star.draw(this.ctx));
        this.drawPaddle(this.ctx);
        this.activeBalls.forEach((ball)=>ball.draw(this.ctx));
        this.activeBricks.forEach((brick)=>brick.draw(this.ctx));
        this.activeGems.forEach((gem)=>gem.draw(this.ctx));
        this.activeParticles.forEach((p)=>p.draw(this.ctx));
        this.activeFloatingTexts.forEach((ft)=>ft.draw(this.ctx));
        if (this.isLevelTransitioning) {
            this.drawLevelCompleteBanner(this.ctx);
        }
        this.ctx.restore();
    }
    drawPaddle(ctx) {
        const rx = Math.round(this.paddle.x - this.paddle.width / 2);
        const ry = Math.round(this.paddle.y);
        const rw = Math.round(this.paddle.width);
        const rh = Math.round(this.paddle.height);
        ctx.save();
        ctx.strokeStyle = '#00f0ff';
        ctx.lineWidth = 3;
        ctx.shadowColor = '#00f0ff';
        ctx.shadowBlur = 10;
        ctx.fillStyle = 'rgba(2, 12, 25, 0.9)';
        ctx.beginPath();
        ctx.roundRect(rx, ry, rw, rh, 6);
        ctx.fill();
        ctx.stroke();
        // Core detail strip
        ctx.strokeStyle = '#ffffff';
        ctx.lineWidth = 1.5;
        ctx.beginPath();
        ctx.moveTo(rx + 20, ry + rh / 2);
        ctx.lineTo(rx + rw - 20, ry + rh / 2);
        ctx.stroke();
        ctx.restore();
    }
    drawLevelCompleteBanner(ctx) {
        ctx.save();
        ctx.fillStyle = 'rgba(5, 2, 10, 0.55)';
        ctx.fillRect(0, 0, this.virtualWidth, this.virtualHeight);
        const cx = this.virtualWidth / 2;
        const cy = this.virtualHeight / 2 - 30;
        const timeSpent = 3.0 - this.levelCompleteTimer;
        if (timeSpent < 1.5) {
            const p = timeSpent / 1.5;
            let scale = 1.0;
            if (p < 0.3) {
                scale = p / 0.3 * 1.3;
            } else if (p < 0.6) {
                scale = 1.3 - (p - 0.3) / 0.3 * 0.4;
            } else {
                scale = 0.9 + (p - 0.6) / 0.9 * 0.1;
            }
            const swayAngle = Math.sin(timeSpent * 8) * 0.03;
            ctx.translate(cx, cy);
            ctx.scale(scale, scale);
            ctx.rotate(swayAngle);
            ctx.shadowColor = '#00f0ff';
            ctx.shadowBlur = 20;
            ctx.fillStyle = '#00f0ff';
            ctx.font = '900 32px "Orbitron", sans-serif';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText(`SECTOR ${this.chunkIndex + 1} CLEARED!`, 0, -20);
            ctx.shadowColor = '#39ff14';
            ctx.shadowBlur = 10;
            ctx.fillStyle = '#39ff14';
            ctx.font = '700 14px "Orbitron", sans-serif';
            ctx.fillText("GRID MODULE OPTIMIZED", 0, 20);
        } else {
            const countdown = Math.max(1, Math.ceil(this.levelCompleteTimer));
            const scale = 1.0 + Math.abs(Math.sin(timeSpent * Math.PI * 2)) * 0.08;
            ctx.translate(cx, cy);
            ctx.scale(scale, scale);
            ctx.shadowColor = '#ffea00';
            ctx.shadowBlur = 15;
            ctx.fillStyle = '#ffea00';
            ctx.font = '900 24px "Orbitron", sans-serif';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText(`INITIALIZING GRID ${this.chunkIndex + 2}`, 0, -20);
            ctx.shadowColor = '#39ff14';
            ctx.shadowBlur = 8;
            ctx.fillStyle = '#39ff14';
            ctx.font = '900 18px "Orbitron", sans-serif';
            ctx.fillText(`LOADING AREA ... ${countdown}`, 0, 18);
        }
        ctx.restore();
    }
    cleanup() {
        if (this.celebrationIntervalId) {
            clearInterval(this.celebrationIntervalId);
            this.celebrationIntervalId = null;
        }
        this.activeBalls.forEach((b)=>this.ballPool.release(b));
        this.activeBricks.forEach((br)=>this.brickPool.release(br));
        this.activeGems.forEach((g)=>this.gemPool.release(g));
        this.activeParticles.forEach((p)=>this.particlePool.release(p));
        this.activeFloatingTexts.forEach((ft)=>this.floatingTextPool.release(ft));
        this.activeBalls = [];
        this.activeBricks = [];
        this.activeGems = [];
        this.activeParticles = [];
        this.activeFloatingTexts = [];
        this.stars = [];
        this.vocabulary = null;
        this.activeChunk = null;
        this.soundManager = null;
    }

    setupExamQuestion(word, options) {
        this.activeBricks = [];
        // Clear and reset ball
        this.activeBalls.forEach((b)=>this.ballPool.release(b));
        this.activeBalls = [];
        this.paddle.x = this.virtualWidth / 2;
        this.spawnBallOnPaddle();
        // Spawn 3 answer bricks at Y = 700
        const optWidth = 160;
        const optHeight = 45;
        const xCoords = [
            30,
            220,
            410
        ];
        options.forEach((opt, idx)=>{
            const brick = this.brickPool.acquire(xCoords[idx], 700, optWidth, optHeight, {
                text: opt.text,
                isCorrect: opt.isCorrect,
                english: word.english,
                turkish: opt.text
            });
            brick.hp = opt.hp;
            brick.maxHp = opt.hp;
            this.activeBricks.push(brick);
        });
    }
    updateExam(dt) {
        if (!window.examEngine || !window.examEngine.isActive) return;
        // Stars update
        this.stars.forEach((star)=>star.update(dt, this.virtualWidth, this.virtualHeight));
        // Paddle keyboard movement
        const paddleSpeed = 500 * this.speedMultiplier;
        if (this.input.left) {
            this.paddle.x -= paddleSpeed * dt;
            this.constrainPaddle();
        }
        if (this.input.right) {
            this.paddle.x += paddleSpeed * dt;
            this.constrainPaddle();
        }
        // Update ball
        for(let i = this.activeBalls.length - 1; i >= 0; i--){
            const ball = this.activeBalls[i];
            ball.update(dt);
            // Wall bounces
            if (ball.x - ball.radius < 0) {
                ball.x = ball.radius;
                ball.vx = -ball.vx;
                this.soundManager.playGemTick();
            }
            if (ball.x + ball.radius > this.virtualWidth) {
                ball.x = this.virtualWidth - ball.radius;
                ball.vx = -ball.vx;
                this.soundManager.playGemTick();
            }
            if (ball.y - ball.radius < 80) {
                ball.y = ball.radius + 80;
                ball.vy = -ball.vy;
                this.soundManager.playGemTick();
            }
            // Paddle bounce
            if (ball.vy > 0 && ball.y + ball.radius >= this.paddle.y && ball.y - ball.radius <= this.paddle.y + this.paddle.height && ball.x + ball.radius >= this.paddle.x - this.paddle.width / 2 && ball.x - ball.radius <= this.paddle.x + this.paddle.width / 2) {
                ball.y = this.paddle.y - ball.radius;
                const relativeHit = (ball.x - this.paddle.x) / (this.paddle.width / 2);
                const maxAngle = Math.PI / 3;
                const angle = relativeHit * maxAngle;
                const speed = 400 * this.speedMultiplier;
                ball.vx = speed * Math.sin(angle);
                ball.vy = -speed * Math.cos(angle);
                this.soundManager.playGemTick();
            }
            // Target word rectangular bounce
            const target = window.examEngine.targetWordRect;
            if (target && target.active && !window.examEngine.inTransition) {
                if (ball.y - ball.radius <= target.y + target.height && ball.y + ball.radius >= target.y && ball.x + ball.radius >= target.x && ball.x - ball.radius <= target.x + target.width) {
                    ball.vy = -ball.vy;
                    this.soundManager.playGemTick();
                }
            }
            // Answer bricks collision
            for(let j = this.activeBricks.length - 1; j >= 0; j--){
                const brick = this.activeBricks[j];
                if (brick.active && checkBallBrickCollision(ball, brick) && !window.examEngine.inTransition) {
                    brick.hp--;
                    if (brick.hp <= 0) {
                        const isCorrect = brick.word.isCorrect;
                        if (isCorrect) {
                            this.soundManager.playExplosion();
                            this.spawnExplosion(brick.x + brick.width / 2, brick.y + brick.height / 2, '#39ff14');
                            brick.active = false;
                            if (!window.examEngine.currentQuestionMissed) {
                                window.examEngine.registerAttempt(window.examEngine.currentWord.english, true);
                                window.examEngine.results.push({
                                    word: window.examEngine.currentWord,
                                    hit: true
                                });
                                window.examEngine.score++;
                            }
                            // Deactivate distractors
                            this.activeBricks.forEach((b)=>{
                                if (b !== brick) {
                                    b.active = false;
                                }
                            });
                            window.examEngine.inTransition = true;
                            window.examEngine.transitionTimer = 1.2;
                            // Proceed next question
                            setTimeout(()=>{
                                window.examEngine.currentIndex++;
                                window.examEngine.updateHUD();
                                window.examEngine.spawnNextWord();
                                // Reset ball for next round
                                this.activeBalls.forEach((b)=>this.ballPool.release(b));
                                this.activeBalls = [];
                                this.paddle.x = this.virtualWidth / 2;
                                this.spawnBallOnPaddle();
                            }, 1200);
                        } else {
                            // Incorrect hit
                            this.soundManager.playDamage();
                            this.spawnExplosion(brick.x + brick.width / 2, brick.y + brick.height / 2, '#ff0055');
                            brick.active = false;
                            if (!window.examEngine.currentQuestionMissed) {
                                window.examEngine.currentQuestionMissed = true;
                                window.examEngine.registerAttempt(window.examEngine.currentWord.english, false);
                                window.examEngine.results.push({
                                    word: window.examEngine.currentWord,
                                    hit: false
                                });
                                window.examEngine.lives--;
                                window.examEngine.updateHUD();
                            }
                        }
                    } else {
                        this.soundManager.playGemTick();
                    }
                    break;
                }
            }
            // Out of bounds drop in Exam
            if (ball.y - ball.radius > 930) {
                this.ballPool.release(ball);
                this.activeBalls.splice(i, 1);
                if (this.activeBalls.length === 0) {
                    if (!window.examEngine.currentQuestionMissed) {
                        window.examEngine.currentQuestionMissed = true;
                        window.examEngine.registerAttempt(window.examEngine.currentWord.english, false);
                        window.examEngine.results.push({
                            word: window.examEngine.currentWord,
                            hit: false
                        });
                        window.examEngine.lives--;
                        window.examEngine.updateHUD();
                    }
                    this.paddle.x = this.virtualWidth / 2;
                    this.spawnBallOnPaddle();
                }
            }
        }
        // Particles update in Exam
        for(let i = this.activeParticles.length - 1; i >= 0; i--){
            const p = this.activeParticles[i];
            p.update(dt);
            if (!p.active) {
                this.particlePool.release(p);
                this.activeParticles.splice(i, 1);
            }
        }
    }

            const ry = Math.round(brick.y);
            const rw = Math.round(brick.width);
            const rh = Math.round(brick.height);
            const isCorrect = brick.word.isCorrect;
            if (isCorrect && window.examEngine.currentQuestionMissed) {
                // Blinking hint theme for correct brick when missed
                const isBlinkOn = Math.floor(performance.now() / 250) % 2 === 0;
                if (isBlinkOn) {
                    ctx.strokeStyle = '#ffea00'; // Neon yellow highlight
                    ctx.lineWidth = 4;
                    ctx.shadowColor = '#ffea00';
                    ctx.shadowBlur = 15;
                    ctx.fillStyle = 'rgba(25, 25, 5, 0.95)';
                } else {
                    ctx.strokeStyle = '#00f0ff';
                    ctx.lineWidth = 2.5;
                    ctx.shadowColor = '#00f0ff';
                    ctx.shadowBlur = 5;
                    ctx.fillStyle = 'rgba(2, 8, 15, 0.85)';
                }
            } else if (isCorrect && window.examEngine.currentWord && window.examEngine.currentWord.isSpecial) {
                ctx.strokeStyle = '#ffea00';
                ctx.lineWidth = 4;
                ctx.shadowColor = '#ffea00';
                ctx.shadowBlur = 15;
                ctx.fillStyle = 'rgba(15, 12, 2, 0.95)';
            } else {
                ctx.strokeStyle = '#00f0ff';
                ctx.lineWidth = 2.5;
                ctx.shadowColor = '#00f0ff';
                ctx.shadowBlur = 10;
                ctx.fillStyle = 'rgba(2, 8, 15, 0.85)';
            }
            ctx.beginPath();
            ctx.roundRect(rx, ry, rw, rh, 8);
            ctx.fill();
            ctx.stroke();
            ctx.shadowBlur = 0;
            const text = brick.word.text;
            const words = text.split(' ');
            let textColor = '#ffffff';
            if (isCorrect && window.examEngine.currentQuestionMissed) {
                const isBlinkOn = Math.floor(performance.now() / 250) % 2 === 0;
                textColor = isBlinkOn ? '#ffea00' : '#ffffff';
            } else if (isCorrect && window.examEngine.currentWord && window.examEngine.currentWord.isSpecial) {
                textColor = '#ffea00';
            }
            ctx.fillStyle = textColor;
            ctx.textAlign = 'center';
            if (words.length > 1) {
                ctx.font = 'bold 11px Arial, sans-serif';
                ctx.textBaseline = 'middle';
                const mid = Math.ceil(words.length / 2);
                const line1 = words.slice(0, mid).join(' ');
                const line2 = words.slice(mid).join(' ');
                ctx.fillText(line1, rx + rw / 2, ry + rh / 2 - 8);
                ctx.fillText(line2, rx + rw / 2, ry + rh / 2 + 8);
            } else {
                let fontSize = 14;
                ctx.font = `bold ${fontSize}px Arial, sans-serif`;
                let textWidth = ctx.measureText(text).width;
                const maxWidth = rw - 12;
                while(textWidth > maxWidth && fontSize > 9){
                    fontSize--;
                    ctx.font = `bold ${fontSize}px Arial, sans-serif`;
                    textWidth = ctx.measureText(text).width;
                }
                ctx.textBaseline = 'middle';
                ctx.fillText(text, rx + rw / 2, ry + rh / 2);
            }
            ctx.restore();
        });
        // Draw active particles in Exam
        this.activeParticles.forEach((p)=>p.draw(ctx));
    }
